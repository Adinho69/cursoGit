-- iniciar um projeto
git init

--checar o que esta adiciondo no controle de versão
git status

--adicionar arquivos ao controle de versão
git add "nome do arquivo"

--adicionar todos os aquivos de uma vez
git add .

--criar uma versao
git commit -m "mensagem para representar oq foi feito nessa versão"

--enviar as alterações para um repositorio na nuvem
git push

--adicionar um repositorio
git remote add origin urldoProjeto

--historico de alterções
git reflog

--voltar a uma versao anterios do codigo, localmente
git reset --hard idDaVersão

--mostrar quais branchs existem
git branch

--criar uma nova branch
git branch nomeDaBranch

--navegar entre as branchs
git checkout nomeDaBranch

--para migrar um codigo para a branch principal, obs tem que estar na branch que deseja importar as alterçoes
git merge nomeDaBranch
